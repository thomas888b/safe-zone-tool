    <script>

        const imageUpload = document.getElementById('imageUpload');

        const imagePreview = document.getElementById('imagePreview');

        const previewContainer = document.getElementById('previewContainer');

        const placeholder = document.getElementById('placeholder');

        

        // All template-related JS variables have been removed


        // Text element

        const instructions = document.getElementById('instructions');



        // Function to handle image upload

        imageUpload.addEventListener('change', (event) => {

            const file = event.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = (e) => {

                    const dataUrl = e.target.result;


                    // Create an in-memory image to check dimensions

                    const img = new Image();

                    img.onload = function() {

                        const imageWidth = this.naturalWidth;

                        const imageHeight = this.naturalHeight;

                        const targetRatio = 16 / 9;

                        const imageRatio = imageWidth / imageHeight;


                        // Check if aspect ratios are very close (e.g., within 0.01)

                        if (Math.abs(imageRatio - targetRatio) < 0.01) {

                            // Ratio is 16:9 - Success

                            imagePreview.src = dataUrl;


                            // Show the preview container and hide the placeholder

                            previewContainer.classList.remove('hidden');

                            placeholder.classList.add('hidden');


                            // Update instructions to normal color

                            instructions.textContent = "Upload another image to check.";

                            instructions.classList.remove('text-red-500');

                            instructions.classList.add('text-gray-400');

                        } else {

                            // Ratio is NOT 16:9 - Error

                            // Hide preview, show placeholder

                            previewContainer.classList.add('hidden');

                            placeholder.classList.remove('hidden');


                            // Show error message

                            instructions.textContent = `Error: Image must be 16:9. (Uploaded: ${imageWidth}x${imageHeight})`;

                            instructions.classList.add('text-red-500');

                            instructions.classList.remove('text-gray-400');

                        }

                    };

                    img.src = dataUrl;

                };

                reader.readAsDataURL(file);

            }

            // Reset file input so the 'change' event fires even

            // if the same file is selected again (e.g., after a failed validation)

            event.target.value = null;

        });


        // All template upload JS functions have been removed

    </script>
